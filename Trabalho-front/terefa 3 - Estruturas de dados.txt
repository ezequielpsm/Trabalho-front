3 - #include <iostream>
#include <cstring>

using namespace std;

int main() {
    char texto[1000];
    char palavra[1000];
    int tam_palavra = 0;

    cout << "Digite um texto terminado com ponto (.):" << endl;
    cin.getline(texto, 1000);

    for (int i = 0; i < strlen(texto); i++) {
        if (texto[i] != ' ' && texto[i] != '.') {
            palavra[tam_palavra++] = texto[i];
        } else {
            for (int j = tam_palavra - 1; j >= 0; j--) {
                cout << palavra[j];
            }
            cout << texto[i];
            tam_palavra = 0;
        }
    }

    cout << endl;

    return 0;
}
